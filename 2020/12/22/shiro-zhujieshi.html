更新login和register的handler

这里需要在原先的项目上重新写登录(用shiro的登录)和注册(用shiro加密)的handler了

package com.chan.shiro.controller;

import com.chan.shiro.pojo.User;
import com.chan.shiro.service.UserService;
import org.apache.shiro.SecurityUtils;
import org.apache.shiro.authc.AuthenticationException;
import org.apache.shiro.authc.IncorrectCredentialsException;
import org.apache.shiro.authc.UnknownAccountException;
import org.apache.shiro.authc.UsernamePasswordToken;
import org.apache.shiro.authz.annotation.Logical;
import org.apache.shiro.authz.annotation.RequiresAuthentication;
import org.apache.shiro.authz.annotation.RequiresPermissions;
import org.apache.shiro.authz.annotation.RequiresRoles;
import org.apache.shiro.subject.Subject;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestMapping;

/**
 * @author ChanServy
 * @date 2019/5/28 19:13
 */
@Controller
@RequestMapping("/user")
public class UserController {

    @Autowired
    UserService userService;

    @RequestMapping("/login/page")
    public String loginPage(){
        return "login";
    }

    @RequestMapping("/login")
    public String login(String username,String password){
        //需要验证权限
        //获取当前用户
        Subject currentUser = SecurityUtils.getSubject();
        UsernamePasswordToken token = new UsernamePasswordToken(username,password);
        //记住我
        token.setRememberMe(true);
        try {
            currentUser.login(token);
            String userPrincipal = (String) currentUser.getPrincipal();
            System.out.println("用户名:"+userPrincipal);
        } catch (UnknownAccountException e){
            //账户不存在 重定向到登录
            System.out.println(e.getMessage());
            return "redirect:login/page";
        } catch (IncorrectCredentialsException ice){
            //密码错误 重定向到登录
            System.out.println(ice.getMessage());
            return "redirect:login/page";
        } catch (AuthenticationException e) {
            //其他错误 重定向到登录
            System.out.println(e.getMessage());
            return "redirect:login/page";
        }
        return "index";
    }

    @RequiresAuthentication
    @RequestMapping("/query")
    public String query(){
        //;用户身份认证了(登陆了)才能查询
        return "index";
    }

    @RequiresAuthentication//登陆了
    @RequiresRoles(value = {"manager","employer"},logical = Logical.OR)//角色是manager或者employer的
    @RequestMapping("/update")
    public String update(){
        //;登陆了 并且角色是manager或者employer的才能访问的路径
        return "index";
    }

    @RequestMapping("/delete")
    @RequiresAuthentication//登陆了
    @RequiresPermissions(value = {"user:save","user:delete"},logical = Logical.AND)//有用户的save和delete的权限的
    public String delete(){
        //;登陆了 并且有用户的save和delete的权限的 才能访问的路径
        return "index";
    }

    //用了shiro的注解 登出的handler就得自己写了
    @RequestMapping("/logout")
    public String logout(){
        Subject subject = SecurityUtils.getSubject();
        subject.logout();
        return "logout";
    }

	
    @RequestMapping("/register")
    public String register(User user){
        userService.insertUser(user);
        return "login";
    }
	//跳到错误页面
    @RequestMapping("/error")
    public String error(){
        return "error";
    }
}

自定义的realm

package com.chan.shiro.realm;

import com.chan.shiro.pojo.User;
import com.chan.shiro.service.UserService;
import org.apache.shiro.authc.*;
import org.apache.shiro.authz.AuthorizationInfo;
import org.apache.shiro.authz.SimpleAuthorizationInfo;
import org.apache.shiro.realm.AuthorizingRealm;
import org.apache.shiro.subject.PrincipalCollection;
import org.apache.shiro.util.ByteSource;

import java.util.Set;

/**
 * @author ChanServy
 * @date 2019/5/29 19:12
 */
public class MyRealm extends AuthorizingRealm {


    UserService userService;

    public void setUserService(UserService userService) {
        this.userService = userService;
    }

    /**
     * 判断 参数中获得的token令牌是否是UsernamePasswordToken类型的
     * @param token
     * @return
     */
    public boolean supports(AuthenticationToken token){
        System.out.println("shiro-realm-type-support");
        if (token instanceof UsernamePasswordToken){
            return true;
        }
        return false;
    }

    /**
     * 角色和权限封装到SimpleAuthorizationInfo
     * @param principalCollection
     * @return
     */
    @Override
    protected AuthorizationInfo doGetAuthorizationInfo(PrincipalCollection principalCollection) {
        System.out.println("角色和权限封装到SimpleAuthorizationInfo......");
        String username = (String) principalCollection.getPrimaryPrincipal();
        System.out.println("用户名:"+username);
//        WebApplicationContext context = ContextLoader.getCurrentWebApplicationContext();
//        UserService userService = (UserService) context.getBean("userServiceImpl");
        System.out.println(username+"的所有角色");
        Set<String> roles = userService.findRoleByUsername(username);
        for (String role : roles) {
            System.out.print(role+"\t");
        }
        System.out.println(username+"的所有权限");
        Set<String> perms = userService.findPermByUsername(username);
        for (String perm : perms) {
            System.out.print(perm+"\t");
        }
        SimpleAuthorizationInfo info = new SimpleAuthorizationInfo();
        info.setRoles(roles);
        info.setStringPermissions(perms);
        return info;
    }

    /**
     * 将认证信息封装到SimpleAuthenticationInfo
     * @param token
     * @return
     * @throws AuthenticationException
     */
    @Override
    protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken token) throws AuthenticationException {
        System.out.println("将身份的认证信息封装到SimpleAuthenticationInfo......");
        String username = (String) token.getPrincipal();
//        WebApplicationContext context = ContextLoader.getCurrentWebApplicationContext();
//        UserService userService = (UserService) context.getBean("userServiceImpl");
        User user = userService.findUserByUsername(username);
        System.out.println("user"+user.toString());
        if (user == null) {
            System.out.println("用户不存在");
            throw new UnknownAccountException("username:"+username+"不存在");
        }else {
            SimpleAuthenticationInfo info = new SimpleAuthenticationInfo(
                    user.getUsername(),
                    user.getPassword(),
                    ByteSource.Util.bytes(user.getSalt()),//将盐值转成字节输入
                    getName());
            return info;
        }
    }
}

UserServiceImpl.java

在之前要先在数据库的用户表加入一列salt盐值 用于加密的时候提高安全性

package com.chan.shiro.service.impl;

import com.chan.shiro.mapper.UserMapper;
import com.chan.shiro.pojo.User;
import com.chan.shiro.service.UserService;
import org.apache.shiro.crypto.hash.Sha256Hash;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.Set;
import java.util.UUID;

/**
 * @author ChanServy
 * @date 2019/5/29 19:19
 */
@Service
public class UserServiceImpl implements UserService {

    @Autowired
    UserMapper userMapper;

    @Override
    public User findUserByUsername(String username) {
        User user = userMapper.findUserByUsername(username);
        return user;
    }

    @Override
    public void insertUser(User user) {
        //生成盐值
        user.setSalt(UUID.randomUUID().toString());
        //用shiro提供的Sha256Hash加密 设置加密属性：sha256算法，随机盐，迭代1024次(和applicationContext-shiro中的配置一致)
        Sha256Hash sha256Hash = new Sha256Hash(user.getPassword(),user.getSalt(),1024);
        //将原先的简单的密码设置为新加密完的密文 并格式化成base64
        user.setPassword(sha256Hash.toBase64());
        //存库
        userMapper.insertUser(user);
    }

    @Override
    public Set<String> findRoleByUsername(String username) {
        Set<String> set = userMapper.findRoleByUsername(username);
        return set;
    }

    @Override
    public Set<String> findPermByUsername(String username) {
        Set<String> set = userMapper.findPermByUserName(username);
        return set;
    }
}

新建applicationContext-shiro.xml文件

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
       xmlns:context="http://www.springframework.org/schema/context"
       xmlns:aop="http://www.springframework.org/schema/aop"
       xmlns:tx="http://www.springframework.org/schema/tx"
       xsi:schemaLocation="http://www.springframework.org/schema/aop http://www.springframework.org/schema/aop/spring-aop-4.3.xsd
		http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans-3.2.xsd
		http://www.springframework.org/schema/tx http://www.springframework.org/schema/tx/spring-tx-4.3.xsd
		http://www.springframework.org/schema/context http://www.springframework.org/schema/context/spring-context-4.3.xsd">
    <!-- shiro组件 -->
    <bean id="myRealm" class="com.chan.shiro.realm.MyRealm">
        <!-- 注入Service -->
        <property name="userService" ref="userServiceImpl"/>
        <!-- 注入密码匹配器 -->
        <property name="credentialsMatcher">
            <bean class="org.apache.shiro.authc.credential.HashedCredentialsMatcher">
                <property name="hashAlgorithmName" value="sha-256"/>
                <property name="hashIterations" value="1024"/>
                <property name="storedCredentialsHexEncoded" value="false"/>
            </bean>
        </property>
        <!-- 开启shiro的缓存 提高性能 -->
        <!-- 默认为false 因为用户的认证信息是放在session中的  session是自动进入缓存的  因此默认为false -->
        <property name="authenticationCachingEnabled" value="false"/>
        <!-- 默认为true 查询出的角色和权限信息封装到info中之后 会放到缓存中 因此默认为true -->
        <property name="authorizationCachingEnabled" value="true"/>
        <!-- 本地缓存组件，存于本地的Map 用户主动logout，或session过期被检测到后，缓存会被移除 -->
        <!--开启shiro的realm的缓存!!!!!!!! 当使用shiro整合redis 使用redis缓存的时候 这里就不用写了 -->
        <!-- name="cacheManager"要写在DefaultWebSecurityManager中 ref到自定义的ShiroRedisCacheManager的id上 -->
        <property name="cacheManager">
            <bean class="org.apache.shiro.cache.MemoryConstrainedCacheManager"/>
        </property>
    </bean>

    <!-- SecurityManager 加注解也保留 -->
    <bean id="securityManager" class="org.apache.shiro.web.mgt.DefaultWebSecurityManager">
        <property name="realm" ref="myRealm"/>
        <property name="rememberMeManager" ref="rememberMeManager"/>
    </bean>
    
    <!-- ShiroFilter -->
    <bean class="org.apache.shiro.spring.web.ShiroFilterFactoryBean" id="shirofilter">
        <!-- 依赖SecurityManager -->
        <property name="securityManager" ref="securityManager"/>
        <!-- 需要登录(认证身份)时 跳转 注解的时候可省略 -->
        <!-- 过滤器链(注解的时候可以省略) -->
    </bean>


    <!--"记住我"的功能-->
    <!-- SimpleCookie帮我们创建cookie并设置相关的参数 -->
    <bean id="rememberMeCookie" class="org.apache.shiro.web.servlet.SimpleCookie">

        <property name="name" value="rememberMe"/>
        <!-- cookie只在http请求中可用，那么通过js脚本将无法读取到cookie信息，有效防止cookie被窃取 -->
        <property name="httpOnly" value="true"/>
        <!-- cookie的生命周期，单位：秒 -->
        <property name="maxAge" value="2592000"/><!-- 30天 -->
    </bean>

    <!--接受SecurityManager调度，获取用户信息，加密数据，并调度SimpleCookie写出cookie-->
    <bean id="rememberMeManager" class="org.apache.shiro.web.mgt.CookieRememberMeManager">

        <property name="cipherKey" value="#{T(org.apache.shiro.codec.Base64).decode('c+3hFGPjbgzGdrC+MHgoRQ==')}"/>
        <!-- 注入SimpleCookie -->
        <property name="cookie" ref="rememberMeCookie"/>
    </bean>

    <!-- SessionManager -->
    <bean id="sessionManager" class="org.apache.shiro.web.session.mgt.DefaultWebSessionManager">
        <!-- session全局超时时间, 单位:毫秒 ，30分钟 默认值为1800000-->
        <property name="globalSessionTimeout" value="5000"/>
        <!--配置session检测-->
        <property name="sessionValidationInterval" value="10000"/>
    </bean>

</beans>

springmvc.xml

注意加上aop的 xml约束 所有的都是在原来项目的基础上添加

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
       xmlns:context="http://www.springframework.org/schema/context"
       xmlns:mvc="http://www.springframework.org/schema/mvc"
       xmlns:aop="http://www.springframework.org/schema/aop"
       xsi:schemaLocation="http://www.springframework.org/schema/mvc http://www.springframework.org/schema/mvc/spring-mvc-4.3.xsd
		http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans-3.2.xsd
		http://www.springframework.org/schema/context http://www.springframework.org/schema/context/spring-context-4.3.xsd
        http://www.springframework.org/schema/aop
        http://www.springframework.org/schema/aop/spring-aop.xsd ">

    <!-- 扫描注解
         通过 【，；】分隔的多个包，包中的类如果有注解，会被扫描
         当然也可以被多个要扫描的包，定义一个父包
    -->
    <context:component-scan base-package="com.chan.shiro.controller"/>

    <!-- 注解开发驱动：识别，解析后端控制器中的注解 -->
    <mvc:annotation-driven></mvc:annotation-driven>

    <!-- 视图解析器
         解析过程：捕获后端控制器的方法返回值，并做前后缀拼接的处理，得到完整的视图路径
     -->
    <bean class="org.springframework.web.servlet.view.InternalResourceViewResolver">
        <property name="prefix" value="/"></property>
        <property name="suffix" value=".jsp"></property>
    </bean>


    <mvc:default-servlet-handler/>


    <!-- enable shiro's annotation 开启shiro的注解模式-->
    <bean id="lifecycleBeanPostProcessor" class="org.apache.shiro.spring.LifecycleBeanPostProcessor"/>
    <!-- 自动代理生成器，等价于aop:config;
         aop:config 或 AutoProxyCreator两者选其一，spring官方提醒千万不要同时使用。

    <bean class="org.springframework.aop.framework.autoproxy.DefaultAdvisorAutoProxyCreator"
          depends-on="lifecycleBeanPostProcessor"/>
    -->
    <aop:config></aop:config>

    <bean class="org.apache.shiro.spring.security.interceptor.AuthorizationAttributeSourceAdvisor">
        <property name="securityManager" ref="securityManager"/>
    </bean>

    <!-- 在使用shiro注解的方式限制权限的时候 身份验证失败了或者角色权限失败了的时候就会报异常 可通过这种方式处理使之跳转到相应期望的页面-->
    <!-- springmvc提供的简单异常处理器 -->
    <bean class="org.springframework.web.servlet.handler.SimpleMappingExceptionResolver">
        <!-- 定义需要特殊处理的异常，这是重要点 -->
        <property name="exceptionMappings">
            <props>
                <!-- Subject 身份认证没通过 跳转的页面 -->
                <prop key="org.apache.shiro.authz.UnauthenticatedException">redirect:/user/login/page</prop>
                <!-- 当前用户权限或者角色验证没通过 跳转的页面 -->
                <prop key="org.apache.shiro.authz.UnauthorizedException">redirect:/user/error</prop>
            </props>
            <!-- 还可以定义其他的自定义异常 -->
        </property>
    </bean>

</beans>

异常处理

我们在用注解的时候 我们是如果身份角色或者权限信息验证失败 正常通过ini或者spring中的配置是会跳到登录页面(身份验证失败时)或者错误页面(没有权限的页面) 但是用了注解之后就不能了 因此要自己制定异常处理

<!-- mvc.xml中：MVC的自定义异常处理器,用于处理权限或身份认证不通过时的异常处理-->
<bean class="com.chan.ex.handler.MyExHandler"></bean>

public class MyExHandler implements HandlerExceptionResolver {
    @Override
    public ModelAndView resolveException(HttpServletRequest request, HttpServletResponse 
                                         response, Object handler, Exception ex) {
        ModelAndView mv = new ModelAndView();
        ex.printStackTrace();
        mv.setViewName("redirect:/login.jsp");
        return mv;
    }
}

在这里 除了自己自定义异常处理器 还可以通过如下的方式来解决, 在springmvc.xml中加入 springmvc提供的简单异常处理器

<!-- springmvc提供的简单异常处理器 -->
    <bean class="org.springframework.web.servlet.handler.SimpleMappingExceptionResolver">
        <!-- 定义需要特殊处理的异常，这是重要点 -->
        <property name="exceptionMappings">
            <props>
                <prop key="org.apache.shiro.authz.UnauthenticatedException">redirect:/user/login/page</prop>
            </props>
            <!-- 还可以定义其他的自定义异常 -->
        </property>
    </bean>
