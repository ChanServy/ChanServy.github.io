WebSocket在线聊天

2020年12月21日
| WebSocket
| 阅读
文章目录

前言：在 http 的协议中，所有的请求都是由客户端发起的，服务端进行响应，但是服务端的话是没办法直接给客户端推送消息的。但是在一些需要即时通信的应用中，我们会有这样的需求，而且这种需求很常见，比如在线游戏、在线股票、支付和最典型的在线聊天等。

WebSocket在线聊天

开头先说说长轮循：请求到达服务器之后我们不急着返回，服务器中如果有数据我们返回，服务器中没数据的话，我们将请求拿着，先不给它返回，这样的话，等到服务器有数据的时候再给它返回，减少无效请求的次数。

但是这样的话有两个问题：

  1. 如果浏览器在服务器响应之前有新的数据要发送，这个时候只能创建一个新的并发请求了。
  2. http下有连接超时的一说，连接上还有的问题就是读取超时，这种长轮循并不能一直持续，会超时的，这样一来会增加程序员的工作量。

下面来说说WebSocket的特点：

  1. WebSocket使用的时候需要创建连接，这样它就变成了一个有状态的协议。
  2. WebSocket使用http协议进行握手。
  3. 维持心跳消息，保持WebSocket一直处在活跃的状态。

public class WebSocketConfig implements WebSocketMessageBrokerConfigurer {

    /**
     * @Description TODO
     * @Param [registry]
     * @return void
     * @Date 2020/6/24 16:19
     * @Author ChenSiYu
     */
    @Override
    public void configureMessageBroker(MessageBrokerRegistry registry) {
        //表示消息代理的前缀,如果后面发的消息的前缀是topic的话,那么这个消息就转发给消息代理,再由消息代理转播给客户端
        registry.enableSimpleBroker("/topic");
        //表示配置一个或者多个前缀,通过这些前缀过滤出需要对具体的方法处理的消息,/app的要用代理处理
        registry.setApplicationDestinationPrefixes("/app");
    }

    /**
     * @Description TODO 连接点
     * @Param [registry]
     * @return void
     * @Date 2020/6/24 16:18
     * @Author ChenSiYu
     */
    @Override
    public void registerStompEndpoints(StompEndpointRegistry registry) {
        registry.addEndpoint("/chat").withSockJS();
    }
}

    function connect() {
        if (!$("#name").val()) {
            return;
        }
        var socket = new SockJS('/chat');//WebSocketConfig配置的chat节点
        stompClient = Stomp.over(socket);
        stompClient.connect({}, function (success) {
            setConnected(true);
            stompClient.subscribe('/topic/chatMany', function (msg) {
                showGreeting(JSON.parse(msg.body));
            });
        })
    }

    function showGreeting(msg) {
        $("#conversation").append('<div>' + msg.name + ':' + msg.content + '</div>');
    }

    function setConnected(flag) {
        $("#connect").prop("disabled", flag);
        $("#disconnect").prop("disabled", !flag);
        if (flag) {
            $("#chat").show();
        } else {
            $("#chat").hide();
        }
    }
</script>
</body>

/**

    private String content;
    private String to;

    public String getFrom() {
        return from;
    }

    public void setFrom(String from) {
        this.from = from;
    }

    public String getContent() {
        return content;
    }

    public void setContent(String content) {
        this.content = content;
    }

    public String getTo() {
        return to;
    }

    public void setTo(String to) {
        this.to = to;
    }
}

/**
     * @Description TODO 在线单聊 点对点的
     * @Param [principal, chat] 
     * @return void
     * @Date 2020/6/24 18:19
     * @Author ChenSiYu
     */
@MessageMapping("/chatOne")//和WebSocketConfig配置的app前缀拼接放在前端的接口
public void chatOne(Principal principal, Chat chat){
    chat.setFrom(principal.getName());//谁登录的这条消息就是从哪来的
    simpMessagingTemplate.convertAndSendToUser(chat.getTo(), "/queue/chatOne", chat);
}

/**
 * @Classname SecurityConfig
 * @Description TODO
 * @Date 2020/6/24 17:58
 * @Created by ChenSiYu
 */
@Configuration
public class SecurityConfig extends WebSecurityConfigurerAdapter {

    @Bean
    PasswordEncoder passwordEncoder() {
        return NoOpPasswordEncoder.getInstance();
    }

    @Override
    protected void configure(AuthenticationManagerBuilder auth) throws Exception {
        auth.inMemoryAuthentication()
                .withUser("陈思禹")
                .password("123")
                .roles("admin")
                .and()
                .withUser("迩东陈")
                .password("123")
                .roles("user");
    }

    @Override
    protected void configure(HttpSecurity http) throws Exception {
        http.authorizeRequests()
                .anyRequest().authenticated()
                .and()
                .formLogin().permitAll();
    }

}
